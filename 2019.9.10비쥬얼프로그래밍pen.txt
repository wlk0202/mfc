대화상자 기반

속성에서 메시지-> WM_MOUSEMOVE 클릭 onmousemove로 전환
 
CPoint pnt;//cpoint 변수
void CpenDlg::OnMouseMove(UINT nFlags, CPoint point)
{
	if (nFlags == MK_LBUTTON) {
		CClientDC dc(this);
		dc.MoveTo(pnt);
		dc.LineTo(point);
	}
	pnt = point;
	CDialogEx::OnMouseMove(nFlags, point);
}
